PROJECT(varfit)

ADD_DEFINITIONS(-DHAVE_INLINE=1 -DGSL_RANGE_CHECK=0)

INCLUDE_DIRECTORIES(../lib_arfit)

ADD_EXECUTABLE(varfit varfit.c)
TARGET_LINK_LIBRARIES(varfit arfit ${VIA_LIBRARY} ${GSL_CBLAS_LIBRARY})

SET_TARGET_PROPERTIES(varfit PROPERTIES
    LINK_FLAGS -Wl)

INSTALL(TARGETS varfit
        RUNTIME DESTINATION bin
        COMPONENT RuntimeLibraries)
